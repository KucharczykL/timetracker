> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add per-user IANA timezone and formatting locale preferences that control wall-clock display and datetime form interpretation without requiring an application restart.

**Architecture:** A dedicated middleware (`TimezoneActivationMiddleware`) activates the resolved timezone on each request and stores the resolved locale on `request._formatting_locale`. The existing `date_time_presentation_for_request()` reads these values. Settings form controls use the existing live-save API (`/api/settings/user/{key}`). Validation uses `zoneinfo.ZoneInfo` for timezones and a hardcoded locale registry.

**Tech Stack:** Python 3.14, Django 5.x, Django Ninja, zoneinfo, PostgreSQL/SQLite

```python
# timetracker/settings_registry.py — add after THEME_CHOICES / _THEME_VALUES

FORMATTING_LOCALE_CHOICES: Final[tuple[tuple[str, str], ...]] = (
    ("cs", "Čeština"),
    ("en-US", "English (US)"),
    ("en-GB", "English (UK)"),
    ("de-AT", "Deutsch (Österreich)"),
    ("de-DE", "Deutsch (Deutschland)"),
    ("pl-PL", "Polski"),
    ("sk-SK", "Slovenčina"),
)
_FORMATTING_LOCALE_VALUES: Final[frozenset[str]] = frozenset(
    value for value, _ in FORMATTING_LOCALE_CHOICES
)


def _validate_formatting_locale(value: object) -> str | None:
    """Validate and normalize a formatting locale value.

    Returns ``None`` for ``None`` input (clearing the field back to unset),
    the stripped value for valid locales, or raises ``ValidationError``.
    """
    if value is None:
        return None
    text = str(value).strip()
    if text not in _FORMATTING_LOCALE_VALUES:
        choices = ", ".join(c[0] for c in FORMATTING_LOCALE_CHOICES)
        raise ValidationError(f"Must be one of: {choices}.")
    return text
```

```python
from zoneinfo import ZoneInfo, ZoneInfoNotFoundError


def _validate_timezone(value: object) -> str:
    """Validate and normalize an IANA timezone value."""
    if isinstance(value, bool):
        raise ValidationError("Timezone must be a string.")
    text = str(value).strip()
    if not text:
        raise ValidationError("Timezone must be a valid IANA zone name.")
    try:
        ZoneInfo(text)
    except ZoneInfoNotFoundError:
        raise ValidationError(f"'{text}' is not a valid IANA timezone.")
    return text
```

```python
    resolve_with_origin,
)


class TimezoneActivationMiddleware:
    """Activate the per-user timezone and store the formatting locale on the request.

    Runs after AuthenticationMiddleware so request.user is available.
    Stores ``request._formatting_locale`` for DateTimePresentation to consume.
    """

    def __init__(self, get_response):
        self.get_response = get_response

    def __call__(self, request):
        user = getattr(request, "user", None)

        if user is not None and user.is_authenticated:
            tz_result = resolve_for_user_with_origin(user, "USER_TIMEZONE")
            locale_result = resolve_for_user_with_origin(
                user, "USER_FORMATTING_LOCALE"
            )

            if isinstance(tz_result.value, str):
                activate(tz_result.value)

            if isinstance(locale_result.value, str):
                request._formatting_locale = locale_result.value
        else:
            # C1 fix: anonymous users resolve DEFAULT_FORMATTING_LOCALE
            # (not USER_FORMATTING_LOCALE — that's USER-scoped and has no
            # site DB row, so resolve_with_origin would skip straight to
            # the default factory, creating confusing indirection).
            locale_result = resolve_with_origin("DEFAULT_FORMATTING_LOCALE")
            if isinstance(locale_result.value, str):
                request._formatting_locale = locale_result.value

        return self.get_response(request)
```

```python
from zoneinfo import ZoneInfo


class IANATimezoneChoices:
    """Iterable of (value, label) pairs from the system's IANA tzdata."""

    _cache: list[tuple[str, str]] | None = None

    @classmethod
    def get(cls) -> list[tuple[str, str]]:
        if cls._cache is None:
            # M2 fix: ZoneInfo.available_timezones() already returns only
            # valid IANA zone names; ZoneInfo(tz) never returns None for
            # these — it raises ZoneInfoNotFoundError.  The dead check is
            # removed.
            cls._cache = sorted(
                (tz, tz) for tz in ZoneInfo.available_timezones()
            )
        return cls._cache


timezones_router = Router()


@timezones_router.get("/", response=list[{"value": str, "label": str, "data": dict}])
def list_timezones(request):
    """Return all available IANA timezone choices for the searchable select."""
    return [
        {"value": tz, "label": tz, "data": {}}
        for tz, _ in IANATimezoneChoices.get()
    ]
```
